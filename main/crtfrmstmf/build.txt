name:
	Create From Stream File (OSSILE)
version:
	0.*
build:
	CRTSRCPF FILE(&INTOLIB/CRTFRMSTMF) RCDLEN(198)
	
	//Copy source
	CPYFRMSTMF FROMSTMF('&DIR/crtfrmstmf.rpgle') TOMBR('/QSYS.lib/&INTOLIB.lib/CRTFRMSTMF.file/RPGLE.mbr') MBROPT(*REPLACE)
	CPYFRMSTMF FROMSTMF('&DIR/crtfrmstmf.pnlgrp') TOMBR('/QSYS.lib/&INTOLIB.lib/CRTFRMSTMF.file/PNLGRP.mbr') MBROPT(*REPLACE)
	CPYFRMSTMF FROMSTMF('&DIR/crtfrmstmf.cmd') TOMBR('/QSYS.lib/&INTOLIB.lib/CRTFRMSTMF.file/CMD.mbr') MBROPT(*REPLACE)
	
	//Create objects
	CRTBNDRPG PGM(&INTOLIB/CRTFRMSTMF) SRCFILE(&INTOLIB/CRTFRMSTMF) SRCMBR(RPGLE) REPLACE(*YES) TEXT('CRTFRMSTMF Program')
	CRTPNLGRP PNLGRP(&INTOLIB/CRTFRMSTMF) SRCFILE(&INTOLIB/CRTFRMSTMF) SRCMBR(PNLGRP) REPLACE(*YES) TEXT('CRTFRMSTMF Panel Group')
	CRTCMD CMD(&INTOLIB/CRTFRMSTMF) PGM(&INTOLIB/CRTFRMSTMF) SRCFILE(&INTOLIB/CRTFRMSTMF) SRCMBR(CMD) TEXT('CRTFRMSTMF Command') HLPPNLGRP(&INTOLIB/CRTFRMSTMF) HLPID(*CMD) REPLACE(*YES)
	
	DLTOBJ OBJ(&INTOLIB/CRTFRMSTMF) OBJTYPE(*FILE)
